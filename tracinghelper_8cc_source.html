133  s += "GL_STENCIL_BUFFER_BIT";
134  mask &= ~GL_STENCIL_BUFFER_BIT;
135  }
137  if (mask)

173  std::ostringstream out;
174 
176  if (int count = base::StringToInt32(type.substr(type.length() - 2, 1))) {
177  int rows = 1;
179  if (type.find("matrix") != std::string::npos)
180  rows = count;
